SELECT *
FROM EMPLOYEE
WHERE SALARY = (SELECT MAX(SALARY) FROM EMPLOYEE);

2. 
WITH subordinates_chain AS (
SELECT ID, CHIEF_ID, 1 AS chain_length
FROM EMPLOYEE
WHERE CHIEF_ID IS NULL
UNION ALL
SELECT E.ID, E.CHIEF_ID, SC.chain_length + 1 AS chain_length
FROM EMPLOYEE E
JOIN subordinates_chain SC ON E.CHIEF_ID = SC.ID
)
SELECT MAX(chain_length) AS max_chain_length
FROM subordinates_chain;

3.
SELECT TOP 1 D.NAME, SUM(E.SALARY) AS TOTAL_SALARY
FROM DEPARTMENT D
JOIN EMPLOYEE E ON D.ID = E.DEPARTMENT_ID
GROUP BY D.ID
ORDER BY TOTAL_SALARY DESC;
